Slideshow

Con los slideshows obtenemos una galeria la cual va variando de imágenes.

Slideshow automatico

Cada dos segundos cambia de imagen, ahora lo discutiremos mas en profundidad.

En esta ocasión no tenemos que añadir los botones, ya que las imágenes pasan automáticamente. Con el ultimo if lo que hacemos es decir el tiepo que pase hasta que el slideshow cambie de imágen. en este ejemplo está puesto a dos segundos.

<div class="w3-content w3-display-container">
<img class="mySlides2" src="f1.jpg" style="width:100%">
<img class="mySlides2" src="f2.jpg" style="width:100%">
<img class="mySlides2" src="f3.jpg" style="width:100%">
<img class="mySlides2" src="f4.jpg" style="width:100%">
</div>
<script>
var slideIndex = 0;
carousel();
function carousel()
{
var i;
var x = document.getElementsByClassName("mySlides2");
for (i = 0; i < x.length; i++)
{
x[i].style.display = "none";
}
slideIndex++;
if (slideIndex > x.length) {slideIndex = 1}
x[slideIndex-1].style.display = "block";
setTimeout(carousel, 2000);
}
</script>

<div class="w3-content w3-display-container">
<img class="mySlides" src="tipo_caminando.jpg" style="width:60%">

<img class="mySlides" src="feroe.jpg" style="width:60%">
<img class="mySlides" src="pedrusco.jpg" style="width:60%">

<a class="w3-btn-floating w3-display-left" onclick="plusDivs(-1)">❮</a>
<a class="w3-btn-floating w3-display-middle" onclick="plusDivs(+1)">❯</a>
</div>

<script>
var slideIndex = 1;
showDivs(slideIndex);

function plusDivs(n) {
showDivs(slideIndex += n);
}

function showDivs(n) {
var i;
var x = document.getElementsByClassName("mySlides");
if (n > x.length) {slideIndex = 1}
if (n < 1) {slideIndex = x.length}
for (i = 0; i < x.length; i++) {
x[i].style.display = "none";
}
x[slideIndex-1].style.display = "block";
}
</script>
